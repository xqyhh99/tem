🔐 三、JWT服务（企业级完整版）
// 🔐 JWT服务（生成AccessToken）
public class JwtService
{
    private readonly IConfiguration _config;

    // 📌 注入配置
    public JwtService(IConfiguration config)
    {
        _config = config;
    }

    // 🚀 生成AccessToken（登录用）
    public string CreateToken(User user, List<string> permissions)
    {
        // 🧠 1. Token携带信息
        var claims = new List<Claim>
        {
            // 🆔 用户ID
            new Claim(ClaimTypes.NameIdentifier, user.Id.ToString()),

            // 👤 用户名
            new Claim(ClaimTypes.Name, user.Username),

            // 🧑‍⚖️ 角色
            new Claim(ClaimTypes.Role, user.Role),

            // 🔐 权限列表
            new Claim("perm", string.Join(",", permissions))
        };

        // 🔑 2. 加密密钥
        var key = new SymmetricSecurityKey(
            Encoding.UTF8.GetBytes(_config["Jwt:Key"]));

        // 🔐 3. 签名算法
        var creds = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);

        // 📦 4. 生成Token
        var token = new JwtSecurityToken(
            issuer: _config["Jwt:Issuer"],
            audience: _config["Jwt:Audience"],
            claims: claims,
            expires: DateTime.Now.AddMinutes(15), // ⏰ AccessToken短期
            signingCredentials: creds
        );

        // 🔄 5. 返回字符串
        return new JwtSecurityTokenHandler().WriteToken(token);
    }
}
🔁 四、RefreshToken（Redis版）
// 🔁 RefreshToken服务（登录续期）
public class RefreshTokenService
{
    private readonly IDatabase _redis;

    // 📌 注入Redis
    public RefreshTokenService(IConnectionMultiplexer redis)
    {
        _redis = redis.GetDatabase();
    }

    // 💾 保存RefreshToken
    public async Task Save(Guid userId, string token)
    {
        await _redis.StringSetAsync(
            $"refresh:{userId}",
            token,
            TimeSpan.FromDays(7)); // ⏰ 7天有效
    }

    // 🔍 获取RefreshToken
    public async Task<string> Get(Guid userId)
    {
        return await _redis.StringGetAsync($"refresh:{userId}");
    }

    // ❌ 删除Token（退出登录）
    public async Task Delete(Guid userId)
    {
        await _redis.KeyDeleteAsync($"refresh:{userId}");
    }
}

// 🔐 登录请求
public record LoginCommand(string Username, string Password)
    : IRequest<LoginResult>;


// 📦 登录返回
public class LoginResult
{
    // 🔐 AccessToken
    public string AccessToken { get; set; }

    // 🔁 RefreshToken
    public string RefreshToken { get; set; }
}


// 🧠 登录逻辑
public class LoginHandler : IRequestHandler<LoginCommand, LoginResult>
{
    private readonly AppDbContext _db;
    private readonly JwtService _jwt;
    private readonly RefreshTokenService _refresh;

    public LoginHandler(AppDbContext db, JwtService jwt, RefreshTokenService refresh)
    {
        _db = db;
        _jwt = jwt;
        _refresh = refresh;
    }

    public async Task<LoginResult> Handle(LoginCommand request, CancellationToken ct)
    {
        // 🔍 查用户
        var user = _db.Users.FirstOrDefault(x => x.Username == request.Username);

        // ❌ 用户不存在
        if (user == null)
            throw new Exception("用户不存在");

        // ❌ 密码错误（生产必须Hash）
        if (user.PasswordHash != request.Password)
            throw new Exception("密码错误");

        // 🔐 权限（真实应来自数据库）
        var permissions = new List<string>
        {
            "user.read",
            "user.create"
        };

        // 🚀 生成AccessToken
        var accessToken = _jwt.CreateToken(user, permissions);

        // 🔁 生成RefreshToken
        var refreshToken = Guid.NewGuid().ToString();

        // 💾 存Redis
        await _refresh.Save(user.Id, refreshToken);

        return new LoginResult
        {
            AccessToken = accessToken,
            RefreshToken = refreshToken
        };
    }
}
🧠 六、RBAC权限中间件
// 🧠 权限中间件（解析JWT权限）
public class PermissionMiddleware
{
    private readonly RequestDelegate _next;

    public PermissionMiddleware(RequestDelegate next)
    {
        _next = next;
    }

    public async Task Invoke(HttpContext context)
    {
        // 🔍 从JWT中读取权限字段
        var perm = context.User.FindFirst("perm")?.Value;

        if (!string.IsNullOrEmpty(perm))
        {
            // ✂️ 存入请求上下文
            context.Items["permissions"] = perm.Split(',');
        }

        await _next(context);
    }
}

    };
});
